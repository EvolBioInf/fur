# Running fur with debian/docker
## Start docker
docker container run --detach-keys="ctrl-@" -h debian -it debian
## Setup system
apt update
apt install build-essential texlive gnuplot libgsl-dev libbsd-dev noweb texlive-pstricks texlive-science texlive-latex-extra ncbi-blas\
t+ git build-essential libdivsufsort-dev cmake dh-autoreconf wget sudo
## Add a user and switch to it
adduser jdoe
adduser jdoe sudo
su jdoe
cd
## Get and install macle
git clone https://github.com/evolbioinf/macle
cd macle
make divsufsort
make sdsl
make
sudo cp build/macle /usr/bin/
cd ../
## Get and install phylonium
git clone https://github.com/evolbioinf/phylonium
cd phylonium
autoreconf -fi -Im4
./configure
make
sudo cp src/phylonium /usr/bin/
cd ../
## Get, build, and test fur
git clone https://github.com/evolbioinf/fur
cd fur
make
make doc
make test
